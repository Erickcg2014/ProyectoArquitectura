CREATE SCHEMA keycloak AUTHORIZATION keycloak;
-- 2️⃣ Crear el schema (sin AUTHORIZATION)
CREATE SCHEMA keycloak;

-- 3️⃣ Dar privilegios de uso y creación dentro del schema
GRANT USAGE, CREATE ON SCHEMA keycloak TO keycloak;

-- 4️⃣ Ajustar el search_path para que trabaje sobre ese schema
ALTER ROLE keycloak SET search_path TO keycloak, public;

-- 5️⃣ Permitir manipulación de tablas y secuencias
ALTER DEFAULT PRIVILEGES IN SCHEMA keycloak
GRANT SELECT, INSERT, UPDATE, DELETE, TRIGGER ON TABLES TO keycloak;

ALTER DEFAULT PRIVILEGES IN SCHEMA keycloak
GRANT USAGE, SELECT, UPDATE ON SEQUENCES TO keycloak;
